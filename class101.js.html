// DESTRUCTURING OBJECTS

/**
 * @author Arhin David
 * Represents a person with various properties.
 * @property {string} name - The name of the person.
 * @property {number} age - The person's age.
 * @property {boolean} isEmployed - Employment status of the person.
 * @property {object} address - An object with city and country info.
 * @property {function} sayHello - A method that returns a greeting message.
 */
const person = {
    name: "Faith",
    age: 25,
    isEmployed: false,
    address: {
        city: "Ho",
        country: "Ghana"
    },
    sayHello: function () {
        return `Hello ${this.name}`;
    },
    hobbies: ["reading", "coding", "cooking"],
}

// Updating object properties
person.age = 29;
person.education = "Masters";

// Example of default values in destructuring
const { schoolName = 'KNUST' } = person;
console.log(schoolName); // 'KNUST' (default value)

// Function demonstrating destructuring in function parameters
function myself({ name, age, isEmployed = "true" }) {
    console.log(`Hello, my name is ${name} and I am ${age} years old. Employment status: ${isEmployed}`);
}
myself(person); // Output: Hello, my name is Faith and I am 29 years old. Employment status: false

// SPREAD OPERATOR
const person2 = { ...person, nickname: "Fay" };
const person3 = { color: 'black' };
const mergeObj = { ...person2, ...person3 };
console.log(mergeObj); // Merges person2 and person3

// Function with rest parameters
function sum(firstNum, secondNum, ...others) {
    console.log(others); // Logs the rest of the numbers passed
    return firstNum + secondNum;
}

const numbers = [1, 2, 3, 4, 5, 6, 7, 9];
console.log(sum(...numbers)); // Spreads the array into the sum function
